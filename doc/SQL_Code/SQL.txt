/* アカウントDB */
CREATE TABLE accounts ( 
    user_id VARCHAR(20) NOT NULL, //ユーザーID
    mail VARCHAR(100) NOT NULL,   //メールアドレス
    pw VARCHAR(20) NOT NULL,  //PW
    nickname VARCHAR(20), //ニックネーム
    pr_group INTEGER, //初期グループ
    PRIMARY KEY  (user_id) 
); 

/* グループDB */
CREATE TABLE groups ( 
    ID INTEGER AUTO_INCREMENT(1), /* 管理番号 #自動採番*/ 
    group_name VARCHAR(20) NOT NULL,  /* グループネーム #自動採番*/ 
    icon VARCHAR(100),　/* アイコン #自動採番*/ 
    PRIMARY KEY  (ID) 
); 

/* グループメンバーDB */
CREATE TABLE groups_memebr ( 
    ID INTEGER AUTO_INCREMENT(1), /* 番号 #自動採番*/ 
    group_ID　INTEGER,
    group_name VARCHAR(20),
    user_id VARCHAR(20),
    editer BOOLEAN,
    PRIMARY KEY  (ID) 
); 

/* タスクDB */
 
create table tasks (
id int not null auto_increment,
group_number int ,
task varchar (30),
contents varchar (200),
deadline date ,
register varchar (20),
to varchar(20),
checkbox boolean ,
manual_link varchar (30)
);

/* 予定DB */
CREATE TABLE schedules ( 
    ID INTEGER AUTO_INCREMENT(1), /* 番号 #自動採番*/ 
    group_number INTEGER,
    task VARCHAR(30) ,
    contents VARCHAR(200),
    today TIMESTAMP,
    register VARCHAR(20),
    PRIMARY KEY  (ID) 
); 

ALTER TABLE SCHEDULES DROP COLUMN TODAY;

/* マニュアルDB */
CREATE TABLE manuals ( 
    id INTEGER AUTO_INCREMENT(1), /* 番号 #自動採番*/ 
    group_number INTEGER,
    manual_name VARCHAR(30),
    PRIMARY KEY  (ID) 
); 

/* 項目DB */
CREATE TABLE items ( 
  id  INTEGER AUTO_INCREMENT(1),
  manual_id	INTEGER,
  heading	VARCHAR(30)		,
  contents  VARCHAR(200)		,
  image VARCHAR(100)		,
  regist_day		Date		,
  PRIMARY KEY(ID)
);
